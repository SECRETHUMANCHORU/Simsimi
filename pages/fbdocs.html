MIRAI/BOTPACK

Chat

module.exports.config = {
    name: "sim",
    version: "1.0.0",
    hasPermission: 0,
    credits: "Choru Official",
    description: "Talk to Sim",
    usages: "[text]",
    commandCategory: "...",
    cooldowns: 5
};

module.exports.run = async ({ api, event, args }) => {
    const { chat } = require('simsimi-sqlite');

    const userInput = args.join(" ");

    try {
        const chatResponse = await chat(userInput, "your-access-token");
        const answer = chatResponse.result;

        api.sendMessage(answer, event.threadID, event.messageID);
    } catch (error) {
        console.error(error);
    }
};

GOATBOT

Chat

module.exports = {
  config: {
    name: "SimSimiChat",
    version: "1.0.0",
    author: "Choru Offcial",
    countDown: 5,
    role: 0,
    shortDescription: {
      vi: "Mô tả ngắn tiếng Việt",
      en: "Short description in English"
    },
    longDescription: {
      uid: "User ID",
      en: "Long description in English"
    },
    category: "Chat",
    guide: {
      vi: "Hướng dẫn sử dụng tiếng Việt",
      en: "Usage guide in English"
    }
  },

  langs: {
    vi: {
      syntaxError: "Lỗi cú pháp"
    },
    en: {
      syntaxError: "Syntax error"
    }
  },

  onStart: async function ({ message, args, getLang }) {
    const { chat } = require('simsimi-sqlite');

    if (!args[0]) {
      const lang = getLang();
      return message.reply(lang.syntaxError);
    }

    try {
      const chatResponse = await chat(args.join(" "), "your-access-token");
      message.reply(`${chatResponse.result}`);
    } catch (error) {
      console.error(error);
      message.reply("An error occurred during the chat.");
    }
  }
};

TEACH


module.exports = {
    config: {
        name: "teach-simsimi",
        version: "1.0.0",
        author: "Choru Official",
        countDown: 5,
        role: 0,
        shortDescription: {
            vi: "Dạy SimSimi",
            en: "Teach SimSimi"
        },
        longDescription: {
            uid: "",
            en: "Teach SimSimi a question and answer pair."
        },
        category: "fun", // Adjust the category as needed
        guide: {
            vi: "Dùng lệnh [text] để dạy SimSimi.",
            en: "Use the [text] command to teach SimSimi."
        }
    },

    langs: {
        vi: {
            syntaxError: "Bạn cần cung cấp cả câu hỏi và câu trả lời để dạy SimSimi."
        },
        en: {
            syntaxError: "You need to provide both a question and an answer to teach SimSimi."
        }
    },

    onStart: async function ({ message, event, args, getLang }) {
        if (args.length < 3 || !args.includes('=>')) {
            const lang = getLang();
            return message.reply(lang.syntaxError);
        }

        const text1 = args.slice(0, args.indexOf('=>')).join(" ");
        const text2 = args.slice(args.indexOf('=>') + 1).join(" ");


        const { teach } = require('simsimi-sqlite');

        try {
            const teachResponse = await teach({
                question: text1,
                answer: text2,
                token: "your-access-token"
            });


            return message.reply(`Teach Response: ${teachResponse.asking}, ${teachResponse.answer}`);
        } catch (error) {
            console.error(error);


            return message.reply('An error occurred while processing the teach request.');
        }
    }
};
